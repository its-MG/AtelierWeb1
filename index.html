Exercice : Programme en C


      #include stdio.h
      
      #define ROWS 3
      #define COLS 3
      
      int main() {
          int matrix[ROWS][COLS];
          int sum = 0;
      
          // Remplissage de la matrice
          printf("Veuillez entrer les éléments de la matrice %dx%d :\n", ROWS, COLS);
          for (int i = 0; i < ROWS; i++) {
              for (int j = 0; j < COLS; j++) {
                  printf("Element [%d][%d] : ", i, j);
                  scanf("%d", &matrix[i][j]);
              }
          }
      
          // Calcul de la somme des éléments positifs
          for (int i = 0; i < ROWS; i++) {
              for (int j = 0; j < COLS; j++) {
                  if (matrix[i][j] > 0) {
                      sum += matrix[i][j];
                  }
              }
          }
      
          // Affichage de la somme
          printf("\nLa somme des éléments positifs de la matrice est : %d\n", sum);
      
          return 0;
      }
